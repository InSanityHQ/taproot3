\subsection{Types of States}
\label{sec:orgee0d03a}
\begin{itemize}
\item Community

\begin{itemize}
\item Ethic group, locale, etc.
\end{itemize}

\item Political entity

\begin{itemize}
\item Community with governmental support
\end{itemize}

\item System

\begin{itemize}
\item A system of 2+ communities with involvement
\end{itemize}

\item Legitimacy

\begin{itemize}
\item A authority point/trust anchor of a community
\end{itemize}
\end{itemize}

\subsection{Statuses of States}
\label{sec:org80a2086}
\begin{itemize}
\item Independent state

\begin{itemize}
\item Political entity w/ ability to make independent decisions both
externally and domestically (note! it could be \emph{influenced} by
others, and still be independent)
\end{itemize}

\item Hegemony

\begin{itemize}
\item A political entity within a system that is able to centrally
\emph{influence} the others' EXTERNAL relations
\end{itemize}

\item Dominion

\begin{itemize}
\item A political entity within a system that is able to centrally
\emph{influence} the other's EXTERNAL and INTERNAL relations
\end{itemize}

\item Empire

\begin{itemize}
\item A political entity within a system that is able to centrally
\emph{control} the other's ENTERNAL and internal RELATIONS, other
political entities in the system are not independent states
\end{itemize}
\end{itemize}
