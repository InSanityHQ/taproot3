\begin{split}
y   &= \frac{a(x+h)^2 -ax^2}{h}
\\  &= \frac{a(x^2+h^2+2xh) -ax^2}{h} 
\\  &= \frac{ax^2 + ah^2 + 2axh -ax^2}{h}
\\  &= \frac{\cancel{ax^2} + axh^2 + 2ah \cancel{- ax^2}}{h}
