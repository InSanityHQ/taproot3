\section{Axler6.23 orthonormal\hfill{}\textsc{def}}
\label{sec:org090fce5}
\begin{quote}
\begin{itemize}
\item A list of vectors is called \emph{orthonormal} if each vector in the list has norm 1 and is orthogonal to all other vectors in the list
\item in other words, a list \(e_1, \ldots, e_m\) of vectors in \(V\) is orthonormal if
\end{itemize}
\[\begin{aligned}
  \langle  e_j, e_k \rangle = \begin{cases}1 & \text{ if } j = k \\ 0 & \text{ if } j \neq  k\end{cases}
  \end{aligned}\]
\end{quote}

\subsection{Axler6.25 norm of an orthonormal linear combination Pythagorean theorem}
\label{sec:orgc3dadfb}
If \(e_1, \ldots, e_m\) is an orthonormal list of vectors in \(V\) then
\[\begin{aligned}
   \lVert a_1 e_1 + \cdots + a_m e_m \rVert ^2 = \vert a_1 \vert ^2 + \cdots + \lvert a_m \rvert^2
   \end{aligned}\]
Pythagorean theorem
\subsection{Axler6.26 orthonormal lists are linearly independent}
\label{sec:orgee112d1}
Any orthogonal list is linearly independent, so an orthonormal list must also be linearly independent.

Or you can use 6.25, set it to zero, and see that all the coefficients must be zero.
