Problem 1

\[ \begin{aligned} KE_{total} &= \sum_{i=1}^{N} \frac{1}{2}m_{i}v_{i}^2 \\ &= \sum_{i=1}^{N} \frac{1}{2}m_{i}(\vec{v}_{i}\cdot \vec{v}_{i}) \\ &= \sum_{i=1}^{N} \frac{1}{2}m_{i}(\vec{V}_{CM}(t)^2 + 2\vec{V}_{CM}(t)\vec{v}_i'(t) + \vec{v}_i'(t)^2) \\ &= \sum_{i=1}^{N} \frac{1}{2}m_{i}(\vec{V}_{CM}(t)^2 + 2\vec{V}_{CM}(t)\vec{v}_i'(t)) + \sum_{i=1}^{N} \frac{1}{2}m_{i}(v_i'^2) \\ &= \frac{1}{2}MV_{CM}^2 + V_{CM}\cdot \sum_{i=1}^{N} m_{i}\vec{v}_i' + \sum_{i=1}^{N} \frac{1}{2}m_{i}(v_i'^2) \\ &= \frac{1}{2}Mv_{CM}^2 + \frac{1}{2} \sum_{i=1}^{N} m_{i}(v_i'^2) \\ \end{aligned} \]
