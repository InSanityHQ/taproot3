\section{Operator, \(\mathcal L(V)\)\hfill{}\textsc{def}}
\label{sec:org7bf484b}
\begin{quote}
\begin{itemize}
\item A linear map from a vector space \(V\) to itself is called an operator.
\item The notation \(\mathcal L(V)\) means \(\mathcal L(V, V)\) which is the set of all operators on \(V\).
\end{itemize}
\end{quote}
