\subsection{concave up}
\label{sec:orgae3dd7e}

\[\begin{aligned}
   \frac{d}{dx} \frac{dy}{dx} = \frac{\dot{x}\ddot{y}-\dot{y}\ddot{x}}{\dot{x}^3} = \frac{2t(6t)-(6t^2-3)(2)}{8t^3}\\
   = \frac{6t^2-6t^2+3}{4t^3} = \frac{3}{4t^3} > 0\\
   \therefore \text{concave up for }t > 0
   \end{aligned}\]
