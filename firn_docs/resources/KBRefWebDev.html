  function reducer(state, action) {
    switch (action.type) { // switch statments with logic, that act on objects passed in!
      case 'increment':
        return {count: state.count + 1};
      case 'decrement':
        return {count: state.count - 1};
      default:
        throw new Error();
    }
  }
